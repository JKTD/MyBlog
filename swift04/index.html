Swift3-04 基础运算符

  • 赋值运算符

    = 这里是没有返回值的。返回值可能是Void，此时不等于nil，即赋值成功；或等于nil，即赋值失败

    var a = 3  
a = 2

  • 数学运算符 + - * / += -= %

  • 逻辑运算符 ! && ||

    var isTrue:Bool = true , isFalse:Bool = false
!isTrue
isTrue && isFalse
isTrue || isFalse

  • 比较运算符 == != > >= < <=

    let money = 100
let price = 20
if money >= price{
    print("buy")
}
    
let capacity = 30 //背包容量
let volume = 40 //购买物体积
    
if money >= price && capacity >= volume{
    print("buy it")
}
    
if money < price || capacity < volume{
    print("can not buy it")
}
    
if  !(money >= price && capacity >= volume){
     print("can not buy it")
}
    
//登陆
var isUserNameOK = false
var isPasswordOk = false
    
var isPhoneNumOk = true
var isPhoneCodeOk = false
    
if isUserNameOK && isPasswordOk || isPhoneNumOk && isPasswordOk{
    print("登陆成功")

    print("登陆失败")
}

    var battery = 19
var batteryColor:UIColor
batteryColor =  battery <= 20 ? UIColor.red:UIColor.green
